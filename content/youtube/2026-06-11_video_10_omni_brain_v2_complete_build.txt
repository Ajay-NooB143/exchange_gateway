import requests, os

def fetch_data(symbol, interval='15min'):
    url = "https://api.twelvedata.com/time_series"
    params = {
        "symbol": symbol,
        "interval": interval,
        "apikey": os.getenv("TWELVEDATA_KEY"),
        "outputsize": 100
    }
    return requests.get(url, params=params).json()
```

def detect_order_blocks(candles):
    obs = []
    for i in range(2, len(candles)):
        prev, curr = candles[i-1], candles[i]
        if prev.close < prev.open and curr.close > curr.open:
            if prev.low < curr.low:  # sweep
                obs.append({

# signal_gen.py
def generate_signal(ob, fvg, divergence, trend):
    score = 0
    if ob: score += 30
    if fvg: score += 20
    if divergence: score += 25
    if trend: score += 25
    
    if score >= 75:
        return {
            'action': 'EXECUTE',
            'score': score,
            'confidence': 'HIGH'
        }
    elif score >= 50:
        return {'action': 'WAIT', 'score': score}
    else:
        return {'action': 'PASS', 'score': score}
```
